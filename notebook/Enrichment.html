import matplotlib.pyplot as plt
import numpy as np

# 计算 -log10(pvalue)
df["neg_log10_p"] = -np.log10(df["adj.P.Val_WT.over.SERT"])

# 定义三类
up = (df["logFC_WT.over.SERT"] > 0.5) & (df["adj.P.Val_WT.over.SERT"] < 0.05)
down = (df["logFC_WT.over.SERT"] < -0.5) & (df["adj.P.Val_WT.over.SERT"] < 0.05)
ns = ~(up | down)  # 非显著

plt.figure(figsize=(6,5))

# 先画不显著（灰色）
plt.scatter(
    df.loc[ns, "logFC_WT.over.SERT"],
    df.loc[ns, "neg_log10_p"],
    color="gray",
    alpha=0.6,
    label="Not significant"
)

# 下调（蓝色）
plt.scatter(
    df.loc[down, "logFC_WT.over.SERT"],
    df.loc[down, "neg_log10_p"],
    color="blue",
    alpha=0.9,
    label="Downregulated"
)

# 上调（红色）
plt.scatter(
    df.loc[up, "logFC_WT.over.SERT"],
    df.loc[up, "neg_log10_p"],
    color="red",
    alpha=0.9,
    label="Upregulated"
)

# 给显著蛋白加标签
for _, row in df.loc[up | down].iterrows():
    plt.text(
        row["logFC_WT.over.SERT"],
        row["neg_log10_p"],
        row["Genes"],
        fontsize=9,
        ha="center",
        va="bottom"
    )
    
# 阈值线
plt.axvline(x=-1, color="grey", linestyle="-")
plt.axvline(x=1, color="grey", linestyle="-")
plt.axhline(y=-np.log10(0.05), color="grey", linestyle="-")

# 对称坐标
max_fc = max(abs(df["logFC_WT.over.SERT"]))
plt.xlim(-max_fc, max_fc)
plt.axvline(0, color='black', linestyle='--')

plt.xlabel("logFC_WT.over.SERT")
plt.ylabel("-log10 adj.P.Val")
plt.title("Volcano Plot")

plt.legend()
plt.tight_layout()

# 导出
plt.savefig(r"C:\Users\cao\Desktop\Python\volcano_plot_WT.over.SERT.png", dpi=300)

plt.show()

import matplotlib.pyplot as plt
import numpy as np
import pandas as pd

# 每个数据库取 top 5，并创建独立副本
top = combined.groupby("Database").head(5).copy()

# 计算显著性分数
top["score"] = -np.log10(top["P-value"])

# 从 Genes 列计算命中基因数
top["gene_count"] = top["Genes"].apply(lambda x: len(str(x).split(";")))

# pathway 名太长时截断
top["Term_short"] = top["Term"].str.slice(0, 60)

plt.figure(figsize=(9, 6))

for db in top["Database"].unique():
    subset = top[top["Database"] == db]
    
    plt.scatter(
        subset["score"],
        subset["Term_short"],
        s=subset["gene_count"] * 80,
        alpha=0.7,
        label=db
    )

plt.xlabel("-log10(P-value)")
plt.ylabel("Pathway / Term")
plt.title("Multi-database Enrichment Bubble Plot")
plt.legend(title="Database")
plt.tight_layout()

plt.savefig("enrichment_bubble_plot.png", dpi=300)
plt.show()
